## 如何调试 Python 项目

```json
{
  // Use IntelliSense to learn about possible attributes.
  // Hover to view descriptions of existing attributes.
  // For more information, visit: https://go.microsoft.com/fwlink/?linkid=830387
  "version": "0.2.0",
  "configurations": [
    {
      "name": "Python APIs",
      "type": "debugpy",
      "request": "launch",
      "module": "uvicorn",
      "cwd": "${workspaceFolder}",
      "args": [
        "yangtze.app.main:app",
        "--reload",
        "--log-level=debug"
        // "--no-access-log"
      ],
      "env": {
        "PYTHONPATH": "${workspaceFolder}"
        // "MONGODB_HOST": "localhost",
        // "MONGODB_PORT": "27017"
      },
      "jinja": true,
      "justMyCode": false
    },
    {
      "name": "Current File",
      "type": "debugpy",
      "request": "launch",
      "program": "${file}",
      "console": "internalConsole",
      "env": {
        "PYTHONPATH": "${workspaceFolder}"
      }
    },
    {
      "name": "Pytest",
      "type": "debugpy",
      "request": "launch",
      "module": "pytest",
      "cwd": "${workspaceFolder}",
      "args": ["tests/api/test_sim_case_set.py", "-vv"]
    }
  ]
}
```
